Python Interview Cheatsheet

Core syntax and concepts for acing the coding interview.

Data Types & Conversion

# Primitives
i = 10        # int
f = 3.14      # float
s = "hi"      # str
b = True      # bool
n = None      # NoneType

# Type Casting
int("123")  # -> 123
str(123)   # -> "123"
float(1)  # -> 1.0
bool(0)   # -> False

Collections Basics

# List (Mutable, Ordered)
l = [1, 2]; l.append(3)

# Tuple (Immutable, Ordered)
t = (1, 2)

# Set (Mutable, No Dups, Unordered)
s = {1, 2}; s.add(3)

# Dict (Mutable, Key-Value)
d = {'a': 1}; d['b'] = 2
d.get('c', 0) # Default value

Comprehensions

# List Comprehension
squares = [x**2 for x in range(5)]
# [0, 1, 4, 9, 16]

evens = [x for x in squares if x % 2 == 0]
# [0, 4, 16]

# Dict Comprehension
sq_dict = {x: x**2 for x in range(3)}
# {0: 0, 1: 1, 2: 4}

String Operations

s = "hello world"
# Slicing [start:stop:step]
s[0:5]  # 'hello'
s[::-1] # 'dlrow olleh' (reverse)

# Formatting
name = "Alice"
f"Hello, {name}!"

# Methods
s.upper()
s.split(' ') # ['hello', 'world']
" ".join(['a','b']) # 'a b'

Functions & Lambdas

def my_func(a, b, *args, **kwargs):
    print(f"a={a}, b={b}")
    print(f"args={args}")
    print(f"kwargs={kwargs}")

my_func(1, 2, 3, 4, c=5, d=6)
# a=1, b=2
# args=(3, 4)
# kwargs={'c': 5, 'd': 6}

# Lambda (anonymous function)
add = lambda x, y: x + y
add(5, 3) # 8

Useful Built-ins

nums = [1, 2, 3]
list(map(lambda x: x*2, nums)) # [2, 4, 6]
list(filter(lambda x: x>1, nums)) # [2, 3]

ids = ['a', 'b']
list(zip(ids, nums)) # [('a', 1), ('b', 2)]

for i, val in enumerate(ids):
    print(i, val) # 0 a, 1 b

Sorting with Key

data = [('A', 3), ('B', 1), ('C', 2)]

# Sort by the second element
sorted(data, key=lambda item: item[1])
# [('B', 1), ('C', 2), ('A', 3)]

# Sort dictionary by value
d = {'a': 2, 'b': 1}
sorted(d.items(), key=lambda kv: kv[1])
# [('b', 1), ('a', 2)]

Common Modules

from collections import defaultdict, Counter
d = defaultdict(int) # default value is 0
c = Counter('aabbc') # {'a':2, 'b':2, 'c':1}

import heapq
h = []; heapq.heappush(h, 3); heapq.heappop(h)

from itertools import combinations
list(combinations([1,2,3], 2)) # [(1,2), (1,3), (2,3)]

from functools import lru_cache
@lru_cache(None) # Memoization
def fib(n): ...

Exception Handling

try:
    result = 10 / 0
except ZeroDivisionError as e:
    print(f"Error: {e}")
except TypeError:
    print("Type mismatch")
else:
    print("No errors occurred.")
finally:
    print("This always runs.")

OOP Basics

class Animal:
    def __init__(self, name):
        self.name = name
    def speak(self):
        pass

class Dog(Animal): # Inheritance
    def speak(self):
        return f"{self.name} says Woof"

my_dog = Dog("Rex")
my_dog.speak() # 'Rex says Woof'

File I/O

# Read from a file
with open('file.txt', 'r') as f:
    content = f.read()
    lines = f.readlines()

# Write to a file
with open('file.txt', 'w') as f:
    f.write("Hello\n")

# Append to a file
with open('file.txt', 'a') as f:
    f.write("World\n")

Advanced Concepts

# Decorator
def my_decorator(func):
    def wrapper():
        print("Before")
        func()
        print("After")
    return wrapper

@my_decorator
def say_hello(): print("Hello")

# Generator
def countdown(n):
    while n > 0:
        yield n
        n -= 1
